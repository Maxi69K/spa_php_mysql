## Create table
CREATE TABLE accounts (
- id INT AUTO_INCREMENT PRIMARY KEY,
- name VARCHAR(50) NOT NULL,
- deposit DECIMAL(10,2) NOT NULL,
- credit_card VARCHAR (50) NOT NULL
- );
